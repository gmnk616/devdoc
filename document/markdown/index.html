import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

```bash
echo Hello Dev
```

</TabItem>

```bash

```

</TabItem>
<TabItem value="prod" label="Prod">

```bash
echo Hello Prod
```

</TabItem>
</Tabs>
